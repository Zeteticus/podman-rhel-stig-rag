FROM registry.access.redhat.com/ubi9/ubi-minimal:latest

ENV _CONTAINERS_USERNS_CONFIGURED=1
ENV PYTHONUNBUFFERED=1

# Install system dependencies
RUN microdnf update -y && \
    microdnf install -y python3 python3-pip python3-devel gcc gcc-c++ && \
    microdnf clean all

WORKDIR /app

# Copy and install requirements
COPY requirements-full.txt /app/
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements-full.txt

# Copy application
COPY app.py /app/
COPY container_start.sh /app/start.sh
RUN chmod +x /app/start.sh

# Create directories
RUN mkdir -p /app/stig_data /app/stig_chroma_db /app/logs /app/templates

EXPOSE 8000

HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:8000/health || exit 1

CMD ["/app/start.sh"]
